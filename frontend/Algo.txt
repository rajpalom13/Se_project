#include <stdio.h>
#include <string.h>
#include <ctype.h>

void removeSpaces(char *str) {
    int i, j = 0;
    for (i = 0; str[i]; i++)
        if (str[i] != ' ')
            str[j++] = str[i];
    str[j] = '\0';
}

void getKeyOrder(char *key, int *order) {
    int len = strlen(key);
    int used[20] = {0};

    for (int i = 0; i < len; i++) {
        int rank = 1;
        for (int j = 0; j < len; j++) {
            if (key[j] < key[i])
                rank++;
            else if (key[j] == key[i] && j < i)
                rank++;
        }
        order[i] = rank;
    }
}

void encrypt(char *plaintext, char *key) {
    removeSpaces(plaintext);

    int plen = strlen(plaintext);
    int klen = strlen(key);

    int rows = (plen + klen - 1) / klen;  // ceiling

    char matrix[20][20];
    int idx = 0;

    // Fill the matrix row-wise
    for (int r = 0; r < rows; r++) {
        for (int c = 0; c < klen; c++) {
            if (idx < plen)
                matrix[r][c] = plaintext[idx++];
            else
                matrix[r][c] = 'X';  // padding
        }
    }

    int order[20];
    getKeyOrder(key, order);

    printf("\nCiphertext: ");

    // Read columns in alphabetical order of key
    for (int num = 1; num <= klen; num++) {
        for (int c = 0; c < klen; c++) {
            if (order[c] == num) {
                for (int r = 0; r < rows; r++)
                    printf("%c", matrix[r][c]);
            }
        }
    }
    printf("\n");
}

void decrypt(char *ciphertext, char *key) {
    int clen = strlen(ciphertext);
    int klen = strlen(key);
    int rows = clen / klen;

    char matrix[20][20];
    int order[20];
    getKeyOrder(key, order);

    int idx = 0;

    // Fill column by column according to key order
    for (int num = 1; num <= klen; num++) {
        for (int c = 0; c < klen; c++) {
            if (order[c] == num) {
                for (int r = 0; r < rows; r++) {
                    matrix[r][c] = ciphertext[idx++];
                }
            }
        }
    }

    printf("\nDecrypted text: ");
    for (int r = 0; r < rows; r++)
        for (int c = 0; c < klen; c++)
            printf("%c", matrix[r][c]);

    printf("\n");
}

int main() {
    char plaintext[100], key[20], ciphertext[100];

    printf("Enter plaintext: ");
    fgets(plaintext, sizeof(plaintext), stdin);
    plaintext[strcspn(plaintext, "\n")] = '\0';

}

def generate_key_matrix(key):
    key = key.upper().replace("J", "I")
    matrix = []
    used = set()

    for char in key:
        if char.isalpha() and char not in used:
            used.add(char)
            matrix.append(char)

    for char in "ABCDEFGHIKLMNOPQRSTUVWXYZ":  # Note: J removed
        if char not in used:
            used.add(char)
            matrix.append(char)

    return [matrix[i:i+5] for i in range(0, 25, 5)]


def find_position(matrix, letter):
    for i in range(5):
        for j in range(5):
            if matrix[i][j] == letter:
                return i, j
    return None


def prepare_text(text):
    text = text.upper().replace(" ", "").replace("J", "I")
    prepared = []

    i = 0
    while i < len(text):
        a = text[i]

        if i + 1 < len(text):
            b = text[i + 1]
            if a == b:
                prepared.append(a + "X")
                i += 1
            else:
                prepared.append(a + b)
                i += 2
        else:
            prepared.append(a + "X")
            i += 1

    return prepared


def encrypt_pair(a, b, matrix):
    r1, c1 = find_position(matrix, a)
    r2, c2 = find_position(matrix, b)

    if r1 == r2:
        # Same row → shift right
        return matrix[r1][(c1 + 1) % 5] + matrix[r2][(c2 + 1) % 5]

    elif c1 == c2:
        # Same column → shift down
        return matrix[(r1 + 1) % 5][c1] + matrix[(r2 + 1) % 5][c2]

    else:
        # Rectangle rule → swap columns
        return matrix[r1][c2] + matrix[r2][c1]


def playfair_encrypt(key, plaintext):
    matrix = generate_key_matrix(key)
    prepared = prepare_text(plaintext)

    encrypted = ""
    for pair in prepared:
        encrypted += encrypt_pair(pair[0], pair[1], matrix)

    return encrypted, matrix

#include <stdio.h>

long long power(long long a, long long b, long long n) {
    long long result = 1;
    while (b > 0) {
        result = (result * a) % n;
        b--;
    }
    return result;
}

int main() {
    // Step 1: Pick small prime numbers
    long long p = 61;
    long long q = 53;

    // Step 2: Compute n and phi(n)
    long long n = p * q;                // 3233
    long long phi = (p - 1) * (q - 1);  // 3120

    // Step 3: Choose public key e (commonly 17)
    long long e = 17;

    // Step 4: Compute private key d such that (d*e) % phi = 1
    long long d = 2753;  // Pre-calculated for simplicity

    // Message
    long long msg = 65; // 'A'

    // Encrypt: c = msg^e % n
    long long encrypted = power(msg, e, n);

    // Decrypt: m = c^d % n
    long long decrypted = power(encrypted, d, n);

    printf("Public Key (e,n)  = (%lld, %lld)\n", e, n);
    printf("Private Key (d,n) = (%lld, %lld)\n", d, n);

    printf("Original Message: %lld\n", msg);
    printf("Encrypted: %lld\n", encrypted);
    printf("Decrypted: %lld\n", decrypted);

    return 0;
}
